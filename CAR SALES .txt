unsigned short getUnsignedShortFromConsole(char message[300]) {

}

char getCharFromConsole(char message[300]) {
	char userInput;
	// the purpose is to save the character that user input
	printf("%s", message);
	// record the character
	scanf("\n%c", &userInput);
	// give back the captured character
	return userInput;
}



void getStringFromConsole(char* message, char* str) {
	// supporting the user
	printf("%s", message);
	// now here this not  required ,rightnow we are on the pointer
	scanf("\n%[^\n]s", str);
}

void swapUnsignedShort(unsigned short* a, unsigned short* b) {
	unsigned short temp;
	// transferring the value discovered from the pointer a's location towards recently formed temp variable
	temp = *a;
	// transferring the value  at the pointer a's location towards the location shown by the pointer b's
	*a = *b;
	// transferring the value of the temp variable towards the location of  pointer b's
	*b = temp;
}

void swapBool(bool* a, bool* b) {
	bool temp;
	// transferring the value discovered at the pointer a's location towards the recently formed temp variable
	temp = *a;
	// transferring the value discovered at the pointer a's  towards the location shown by the pointer b's
	*a = *b;
	// transferring the temp variable's value to the location of pointer b's
	*b = temp;
}

FILE* createFile(char fileName[201]) {
	// file is opening to write and update
	file = fopen(fileName, "w");
	if (file != NULL) {
		fclose(file);
	}
	return file;
}


void openFile(char fileName[201], char mode[4]) {
	// trying to open the file
	file = fopen(fileName, mode);

	// when value become null
	if (file == NULL) {

		// first we  create a file and check whether it is null
		// this indicates that something went wrong
		if (createFile(fileName) == NULL) {
			fileStatus = FILE_ERROR;

			//  show error in opening the file
			printf("Error opening %s: %s", fileName, strerror(errno));
		}
		else {
			// now we are opening the file
			openFile(fileName, mode);
		}

	}
	// if nothing went wrong file opens
	else {
		fileStatus = FILE_OPENED;
	}
}

// once we open a file we need to close a file too
void closeFile() {
	// if the file is opened then need to close
	if (fileStatus == FILE_OPENED) {
		fclose(file);
        //then file close here
		fileStatus = FILE_CLOSED;
	}
}


void readDataFromFile() {
	//read the lines from the file
	int lineCounter = 0;

	// an indefinite loop that will be manually terminated if we reach at the bottom of the file
	while (1) {

		unsigned short carAmountPerSaleValue = 0, typeOfcarPerSaleValue = 0, discountGivenPerSaleValue = 0;
		char customerNameValue[201] = "";

		int scanResult = fscanf(

			file,
			"%hd,%hd,%hd,%[^\n]s",
			&carAmountPerSaleValue, // the variables
			&typeOfcarPerSaleValue,
			&discountGivenPerSaleValue,
			customerNameValue
		);

		//  when arrive at the bottom of the  file
		if (scanResult == EOF) {
			// end for the loop
			break;
		}


		carAmountPerSale[lineCounter] = carAmountPerSaleValue;
		typeOfCarPerSale[lineCounter] = typeOfcarPerSaleValue;
		//placing the discount per sales from array converting to given per sales value
		discountGivenPerSale[lineCounter] = (bool)discountGivenPerSaleValue;
		// because we are dealing sting we need to utilize strcpy
		strcpy(consumerNames[lineCounter], customerNameValue);

		// scan the upcoming line
		lineCounter++;
	}

	// aware of sales that are available following the preceding procedure
	numberOfSales = lineCounter;
}

void getDataFromFile() {
	openFile(CSV_FILE, "r");

	if (fileStatus == FILE_OPENED) {
		readDataFromFile();
	}
	else if (fileStatus == FILE_ERROR) {
		printf("When attempting to read from the file, an error occurred. %s.", CSV_FILE);
		// the programming is not running due to error in the file
		pauseProgram('_');
	}

	closeFile();
}

void writeDataToFile() {
	// loop through
	for (int i = 0; i < numberOfSales; i++) {

		char line[201];
		char data[50];

		//changing the data string to variable into string and putting unsigned short value in the data string variable
        //a typical numerical value 10 indicate base 10
		_itoa((int)carAmountPerSale[i], data, 10);
		//now we are using strcpy and including the rate of cars in a line
		//		the main  use strcat  is to include more to the string
		strcpy(line, data);
		// to split the value we include a comma
		strcat(line, ",");

		// this value is transforming into string
		_itoa((int)typeOfCarPerSale[i], data, 10);
		// include  the kind of car to the line
		strcat(line, data);
		// to split the value we include a comma
		strcat(line, ",");

		// this value is transforming into string
		_itoa((int)discountGivenPerSale[i], data, 10);
		// including the discount
		strcat(line, data);
		// to split the value we include a comma
		strcat(line, ",");

		// including the name of consumer
		strcat(line, consumerNames[i]);

		// create a line into the file
		fprintf(file, "%s", line);

		// when we not reach at the final we use a newline character
		if (i < numberOfSales - 1) {
			fprintf(file, "\n");
		}
	}
}

void saveDataToFile() {
	openFile(CSV_FILE, "w");

	if (fileStatus == FILE_OPENED) {
		writeDataToFile();

	else if (fileStatus == FILE_ERROR) {
		printf("When attempting to read from the file, an error occurred %s.", CSV_FILE);
		//this tells it we're using it due to a file error
		pauseProgram('_');
	}

	closeFile();
}

    float menu_applyDiscount(float currentPrice) {
        //applying discount and reducing the price (0.05) from the number of car price given from array
        return currentPrice * (1 - DISCOUNT_PERCENTAGE);
    }

    bool menu_checkIfDiscountIsNeeded(unsigned short userAge) {
        // giving discount if they have discount
        if (userAge >= DISCOUNT_MIN_AGE && userAge <= DISCOUNT_MAX_AGE) {
            return TRUE;
        } else {
            //  checking whether user has unidays or license , check if they have we need to give the discount
            char nusCardResponse = getCharFromConsole("\nDo you have an unidays or license ? Answer 'y' or 'n' : ");
            //if y giving discount
            if (nusCardResponse == 'y') {
                return TRUE;
            }
        }

        // return the FALSE value to indicate that no discount is to be given
        return FALSE;
    }

    }

        void sortArraysByNumberOfCarsSoldPerSale() {

            // provide a loop that  help to to get a index for accessing the a value  of first and second
            for (int i = 0; i < numberOfSales - 1; i++) {

                // we creating a loop that helps us to  access the index j between first at that time and  a places which contain the value
                //this loop is to run completely from start to end
                for (int j = i + 1; j < numberOfSales; j++) {

                    // looking the location i of value if the value is greater than the value at the  location j
                    if (carAmountPerSale[i] > carAmountPerSale[j]) {

                        //  changing two value at array called the carAmountPerSale
                        swapUnsignedShort(&carAmountPerSale[i], &carAmountPerSale[j]);

                        //  changing two value at the array typeofCarSale in the  same position
                        swapUnsignedShort(&typeOfCarPerSale[i], &typeOfCarPerSale[j]);

                        //  discountGivenPerSale array
                        swapBool(&discountGivenPerSale[i], &discountGivenPerSale[j]);
                        //  customerNames array

                        // doing changes completely
                        char temp[201];
                        // string is transferred from position i to  the newly created temp variable
                        strcpy(temp, consumerNames[i]);
                        //  string is transferred from position j to position i
                        strcpy(consumerNames[i], consumerNames[j]);
                        // string is transferred from temp to position j
                        strcpy(consumerNames[j], temp);

                    }

                }

            }

        }

        void printSalesDataAtPosition(int position) {

            //in here we are utilising the value explored at the position i in the typeOfCarPerSale array
            // carPrice array is to determine the rate of the type of the car
            int typeOfCar = typeOfCarPerSale[position];
            // evaluate the similar sale's price
            float price = carAmountPerSale[position] * carRates[typeOfCar];

            // evaluate the display the discount text for the user's advantages
            char discountGivenText[4];
            // if we give a discount
            if (discountGivenPerSale[position] == TRUE) {
                // if yes show the text
                strcpy(discountGivenText, "Yes");
                // change the show text yes
                price *= (1 - DISCOUNT_PERCENTAGE);
            } else {
                // change the show text no
                strcpy(discountGivenText, "No");
            }

            // display the information that user input to each sale
            // the main reason to breaking down the printf() command to many lines to make it easier to understand
            printf("Sale Index: %hd | Sale Rate: %f | Name of cars: %s | "
                   "Car Price: %d | Number of Cars: %hd | "
                   "Discount Given: %s | Consumer Name: %s|Consumer Number:1\n",
                    // showing that above given strings that will merged by the programme
                   position, price, carNames[typeOfCar],
                   carRates[typeOfCar], carAmountPerSale[position],
                   discountGivenText, consumerNames[position]);

        }


        void menu_viewSales() {
            // before we showing the data of the sale we need to sort the array
            sortArraysByNumberOfCarsSoldPerSale();

            /* using the variable in menu_viewSales  */
            //  total sale price and total number of cars sold these two will contain
            float totalSalesValue = 0;
            unsigned int carsSold = 0;

            printf("All Sales Data:\n\n");

            // making the  loop that helps to run the code many time
            // variable that show numberOfSales
            for (int i = 0; i < numberOfSales; i++) {

                //in here we uses the value we dicovered at the position i in the typeOfCarPerSale
                //we know that the position of the carRate array will determine the rate of the type of car
                int typeOfCar = typeOfCarPerSale[i];
                // evaluate  the current sale's rate
                float price = carAmountPerSale[i] * carRates[typeOfCar];

                // this function is to print the row in sales
                printSalesDataAtPosition(i);

                // including total number of cars sold
                totalSalesValue += price;
                carsSold += carAmountPerSale[i];

            }

            // showing the overall data
            printf("\n%hd cars have been sold with a total value of %f GBP. There are %hd cars unsold and are in the showroom.\n",
                   carsSold, totalSalesValue, carsintheShowroom);
        }

        void printSalesDataBasedOnCarsSoldBetweenMinMax(unsigned short min, unsigned short max) {

            printf("Sales Data containing between %hd and %hd cars sold:\n\n", min, max);

            // finding the array to take a look in index
            //	the value will contain in starting with 0 and ending the last position
            //	checking every statement in every value
            //	finding a value that made a relation between certain rules that means we found it
            //		value in array is found

            //checking through all location that is stored in a value in our carAmountPerSale array
            for (int i = 0; i < numberOfSales; i++) {

                // checking the value is min or max
                if (min <= carAmountPerSale[i] && carAmountPerSale[i] <= max) {

                    // printing  the sales data
                    printSalesDataAtPosition(i);

                }

            }

        }
